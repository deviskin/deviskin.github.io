וַיַּעַן־יְהוָה אֶת־אִיּוֹב מִן (מִן) הַסְּעָרָה (הַסְּעָרָה), וַיֹּאמַר׃
מִי זֶה מַחְשִׁיךְ עֵצָה בְמִלִּין, בְּלִי־דָעַת׃
אֱזָר־נָא כְגֶבֶר חֲלָצֶיךָ; וְאֶשְׁאָלְךָ, וְהוֹדִיעֵנִי׃
אֵיפֹה הָיִיתָ בְּיָסְדִי־אָרֶץ; הַגֵּד, אִם־יָדַעְתָּ בִינָה׃
מִי־שָׂם מְמַדֶּיהָ כִּי תֵדָע; אוֹ מִי־נָטָה עָלֶיהָ קָּו׃
עַל־מָה אֲדָנֶיהָ הָטְבָּעוּ; אוֹ מִי־יָרָה, אֶבֶן פִּנָּתָהּ׃
בְּרָן־יַחַד כּוֹכְבֵי בֹקֶר; וַיָּרִיעוּ, כָּל־בְּנֵי אֱלֹהִים׃
וַיָּסֶךְ בִּדְלָתַיִם יָם; בְּגִיחוֹ, מֵרֶחֶם יֵצֵא׃
בְּשׂוּמִי עָנָן לְבֻשׁוֹ; וַעֲרָפֶל, חֲתֻלָּתוֹ׃
וָאֶשְׁבֹּר עָלָיו חֻקִּי; וָאָשִׂים, בְּרִיחַ וּדְלָתָיִם׃
וָאֹמַר, עַד־פֹּה תָבוֹא וְלֹא תֹסִיף; וּפֹא־יָשִׁית, בִּגְאוֹן גַּלֶּיךָ׃
הֲמִיָּמֶיךָ צִוִּיתָ בֹּקֶר; יִדַּעְתָּה (יִדַּעְתָּה) שַׁחַר (הַשַּׁחַר) מְקֹמוֹ׃
לֶאֱחֹז בְּכַנְפוֹת הָאָרֶץ; וְיִנָּעֲרוּ רְשָׁ<סuפ>עִסuפ>ים מִמֶּנָּה׃
תִּתְהַפֵּךְ כְּחֹמֶר חוֹתָם; וְיִתְיַצְּבוּ, כְּמוֹ לְבוּשׁ׃
וְיִמָּנַע מֵרְשָׁ<סuפ>עִסuפ>ים אוֹרָם; וּזְרוֹעַ רָמָה, תִּשָּׁבֵר׃
הֲבָאתָ עַד־נִבְכֵי־יָם; וּבְחֵקֶר תְּהוֹם, הִתְהַלָּכְתָּ׃
הֲנִגְלוּ לְךָ שַׁעֲרֵי־מָוֶת; וְשַׁעֲרֵי צַלְמָוֶת תִּרְאֶה׃
הִתְבֹּנַנְתָּ עַד־רַחֲבֵי־אָרֶץ; הַגֵּד, אִם־יָדַעְתָּ כֻלָּהּ׃
אֵי־זֶה הַדֶּרֶךְ יִשְׁכָּן־אוֹר; וְחֹשֶׁךְ, אֵי־זֶה מְקֹמוֹ׃
כִּי תִקָּחֶנּוּ אֶל־גְּבוּלוֹ; וְכִי־תָבִין, נְתִיבוֹת בֵּיתוֹ׃
יָדַעְתָּ כִּי־אָז תִּוָּלֵד; וּמִסְפַּר יָמֶיךָ רַבִּים׃
הֲבָאתָ אֶל־אֹצְרוֹת שָׁלֶג; וְאֹצְרוֹת בָּרָד תִּרְאֶה׃
אֲשֶׁר־חָשַׂכְתִּי לְעֶת־צָר; לְיוֹם קְרָב, וּמִלְחָמָה׃
אֵי־זֶה הַדֶּרֶךְ יֵחָלֶק אוֹר; יָפֵץ קָדִים עֲלֵי־אָרֶץ׃
מִי־פִלַּג לַשֶּׁטֶף תְּעָלָה; וְדֶרֶךְ, לַחֲזִיז קֹלוֹת׃
לְהַמְטִיר עַל־אֶרֶץ לֹא־אִישׁ; מִדְבָּר, לֹא־אָדָם בּוֹ׃
לְהַשְׂבִּיעַ שֹׁאָה וּמְשֹׁאָה; וּלְהַצְמִיחַ, מֹצָא דֶשֶׁא׃
הֲיֵשׁ־לַמָּטָר אָב; אוֹ מִי־הוֹלִיד, אֶגְלֵי־טָל׃
מִבֶּטֶן מִי יָצָא הַקָּרַח; וּכְפֹר שָׁמַיִם מִי יְלָדוֹ׃
כָּאֶבֶן מַיִם יִתְחַבָּאוּ; וּפְנֵי תְהוֹם, יִתְלַכָּדוּ׃
הַתְקַשֵּׁר מַעֲדַנּוֹת כִּימָה; אוֹ־מֹשְׁכוֹת כְּסִיל תְּפַתֵּחַ׃
הֲתֹצִיא מַזָּרוֹת בְּעִתּוֹ; וְעַיִשׁ, עַל־בָּנֶיהָ תַנְחֵם׃
הֲיָדַעְתָּ חֻקּוֹת שָׁמָיִם; אִם־תָּשִׂים מִשְׁטָרוֹ בָאָרֶץ׃
הֲתָרִים לָעָב קוֹלֶךָ; וְשִׁפְעַת־מַיִם תְּכַסֶּךָּ׃
הַתְשַׁלַּח בְּרָקִים וְיֵלֵכוּ; וְיֹאמְרוּ לְךָ הִנֵּנוּ׃
מִי־שָׁת בַּטֻּחוֹת חָכְמָה; אוֹ מִי־נָתַן לַשֶּׂכְוִי בִינָה׃
מִי־יְסַפֵּר שְׁחָקִים בְּחָכְמָה; וְנִבְלֵי שָׁמַיִם, מִי יַשְׁכִּיב׃
בְּצֶקֶת עָפָר לַמּוּצָק; וּרְגָבִים יְדֻבָּקוּ׃
הֲתָצוּד לְלָבִיא טָרֶף; וְחַיַּת כְּפִירִים תְּמַלֵּא׃
כִּי־יָשֹׁחוּ בַמְּעוֹנוֹת; יֵשְׁבוּ בַסֻּכָּה לְמוֹ־אָרֶב׃
מִי יָכִין לָעֹרֵב, צֵידוֹ כִּי־יְלָדוֹ (יְלָדָיו) אֶל־אֵל יְשַׁוֵּעוּ; יִתְעוּ, לִבְלִי־אֹכֶל׃
