*55:1לַמְנַצֵּחַ בִּנְגִינֹת, מַשְׂכִּיל לְדָוִד׃
55:2הַאֲזִינָה אֱלֹהִים תְּפִלָּתִי; וְאַל־תִּתְעַלַּם, מִתְּחִנָּתִי׃
55:3הַקְשִׁיבָה לִּי וַעֲנֵנִי; אָרִיד בְּשִׂיחִי וְאָהִימָה׃
55:4מִקּוֹל אוֹיֵב, מִפְּנֵי עָקַת רָשָׁע; כִּי־יָמִיטוּ עָלַי אָוֶן, וּבְאַף יִשְׂטְמוּנִי׃
55:5לִבִּי יָחִיל בְּקִרְבִּי; וְאֵימוֹת מָוֶת, נָפְלוּ עָלָי׃
55:6יִרְאָה וָרַעַד יָבֹא בִי; וַתְּכַסֵּנִי, פַּלָּצוּת׃
55:7וָאֹמַר, מִי־יִתֶּן־לִּי אֵבֶר כַּיּוֹנָה, אָעוּפָה וְאֶשְׁכֹּנָה׃
55:8הִנֵּה אַרְחִיק נְדֹד; אָלִין בַּמִּדְבָּר סֶלָה׃
55:9אָחִישָׁה מִפְלָט לִי; מֵרוּחַ סֹעָה מִסָּעַר׃
55:10בַּלַּע אֲדֹנָי פַּלַּג לְשׁוֹנָם; כִּי־רָאִיתִי חָמָס וְרִיב בָּעִיר׃
55:11יוֹמָם וָלַיְלָה, יְסוֹבְבֻהָ עַל־חוֹמֹתֶיהָ; וְאָוֶן וְעָמָל בְּקִרְבָּהּ׃
55:12הַוּוֹת בְּקִרְבָּהּ; וְלֹא־יָמִישׁ מֵרְחֹבָהּ, תֹּךְ וּמִרְמָה׃
55:13כִּי לֹא־אוֹיֵב יְחָרְפֵנִי, וְאֶשָּׂא לֹא־מְשַׂנְאִי עָלַי הִגְדִּיל; וְאֶסָּתֵר מִמֶּנּוּ׃
55:14וְאַתָּה אֱנוֹשׁ כְּעֶרְכִּי; אַלּוּפִי, וּמְיֻדָּעִי׃
55:15אֲשֶׁר יַחְדָּו נַמְתִּיק סוֹד; בְּבֵית אֱלֹהִים, נְהַלֵּךְ בְּרָגֶשׁ׃
55:16יַשִּׁימָוֶת (יַשִּׁי מָוֶת) עָלֵימוֹ, יֵרְדוּ שְׁאוֹל חַיִּים; כִּי־רָעוֹת בִּמְגוּרָם בְּקִרְבָּם׃
55:17אֲנִי אֶל־אֱלֹהִים אֶקְרָא; וַיהוָה, יוֹשִׁיעֵנִי׃
55:18עֶרֶב וָבֹקֶר וְצָהֳרַיִם אָשִׂיחָה וְאֶהֱמֶה; וַיִּשְׁמַע קוֹלִי׃
55:19פָּדָה בְשָׁלוֹם נַפְשִׁי מִקֲּרָב־לִי; כִּי־בְרַבִּים, הָיוּ עִמָּדִי׃
55:20יִשְׁמַע אֵל וְיַעֲנֵם וְיֹשֵׁב קֶדֶם, סֶלָה אֲשֶׁר אֵין חֲלִיפוֹת לָמוֹ; וְלֹא יָרְאוּ אֱלֹהִים׃
55:21שָׁלַח יָדָיו בִּשְׁלֹמָיו, חִלֵּל בְּרִיתוֹ׃
55:22חָלְקוּ מַחְמָאֹת פִּיו וּקֲרָב־לִבּוֹ רַכּוּ דְבָרָיו מִשֶּׁמֶן, וְהֵמָּה פְתִחוֹת׃
55:23הַשְׁלֵךְ עַל־יְהוָה יְהָבְךָ וְהוּא יְכַלְכְּלֶךָ לֹא־יִתֵּן לְעוֹלָם מוֹט, לַצַּדִּיק׃
55:24וְאַתָּה אֱלֹהִים תּוֹרִדֵם לִבְאֵר שַׁחַת, אַנְשֵׁי דָמִים וּמִרְמָה לֹא־יֶחֱצוּ יְמֵיהֶם; וַאֲנִי, אֶבְטַח־בָּךְ׃
