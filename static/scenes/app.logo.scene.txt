если (режим == "запуск") {
  память.классПриложения = класс {
    d3 = загрузитьПоддержкуD3();

    высота = 128;
    ширина = 512;

    выполнить() {
      текст = текстЛоготипа;
      высотаШрифта = высота/10*6;
      мт = d3.метрикаТекста(текст, высотаШрифта, ложь);
      вертПоз = 0;
      если (мт.ширина > ширина) {
        высотаШрифта = высотаШрифта * ширина / мт.ширина;
        мт2 = d3.метрикаТекста(текст, высотаШрифта, ложь);
        вертПоз += (высота - мт2.высота) / 2;
      } иначе если (высота > мт.высота) {
        вертПоз += (высота - мт.высота) / 2;
      }
      вертПоз += высотаШрифта;

      d3.текст()
        .текст(текст)
        .размер(высотаШрифта)
        .текстовыйЯкорь('middle')
        .позиция(ширина/2,вертПоз)
        .заливка('lightgray')
        .прозрачность(0)
        .контур('none')
        .переход()
        .продолжительность(1000)
        .прозрачность(1)
        ;
      d3.установитьРазмерХолста(ширина,высота);
    }
  };
  память.приложение = память.классПриложения();
}

если (режим == "запуск") {
  память.приложение.выполнить();
}
